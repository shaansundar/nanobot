```python
@dataclass
class LLMResponse:
    content: str | None
    tool_calls: list[ToolCallRequest] = field(default_factory=list)
    finish_reason: str = "stop"
    usage: dict[str, int] = field(default_factory=dict)
    error_status_code: int | None = None
    error_kind: str | None = None
    error_type: str | None = None
    error_code: str | None = None
    error_retry_after_s: float | None = None
    error_should_retry: bool | None = None

class LLMProvider(ABC):
    def __init__(self, api_key: str | None = None, api_base: str | None = None):
        ...
    @abstractmethod
    async def chat(
        self,
        messages: list[dict[str, Any]],
        tools: list[dict[str, Any]] | None = None,
        model: str | None = None,
        max_tokens: int = 4096,
        temperature: float = 0.7,
        reasoning_effort: str | None = None,
        tool_choice: str | dict[str, Any] | None = None,
    ) -> LLMResponse:
        ...
    @abstractmethod
    def get_default_model(self) -> str:
        ...
```

  - List args (never shell=True) per D-02
- Drain stdout/stderr concurrently per D-04: `stdout_bytes, stderr_bytes = await proc.communicate()`
- `await proc.wait()`
- Return `self._parse_result(stdout_bytes, stderr_bytes, proc.returncode)`

`_extract_latest_user_content(self, messages: list[dict[str, Any]]) -> str`:
- Iterate `reversed(messages)`, find first `role == "user"`
- If content is str, return it
- If content is list, join text blocks: `[block["text"] for block in content if block.get("type") == "text"]`
- Return `""` if no user message found

`_parse_result(self, stdout_bytes: bytes, stderr_bytes: bytes, returncode: int | None) -> LLMResponse`:
- Decode stderr, log at debug level if non-empty: `logger.debug("Claude Code stderr: {}", stderr_text[:500])`
- Decode stdout. If empty: return error LLMResponse with content `f"Claude Code CLI returned no output. Exit code: {returncode}. stderr: {stderr_text[:200]}"`

- If `is_error is True`: return `self._build_error_response(result_text, stderr_text, usage_dict)`
- If success: return `LLMResponse(content=result_text or None, tool_calls=[], finish_reason="stop", usage={"prompt_tokens": usage.get("input_tokens", 0), "completion_tokens": usage.get("output_tokens", 0)})` (per D-17, tool_calls always empty)

`_build_error_response(self, result_text: str, stderr_text: str, usage: dict) -> LLMResponse`:

  - Default: `error_should_retry=False`, `error_kind="cli_error"`
- Return `LLMResponse(content=result_text, tool_calls=[], finish_reason="error", usage={"prompt_tokens": usage.get("input_tokens", 0), "completion_tokens": usage.get("output_tokens", 0)}, error_kind=error_kind, error_should_retry=error_should_retry)`
