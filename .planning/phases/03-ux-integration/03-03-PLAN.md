    **Fix 1 -- onboard.py get_provider_choices (per D-13, D-15):**

    In `_configure_providers()`, replace the `get_provider_choices()` inner function. The current code unconditionally accesses `provider.api_key` which crashes for `ClaudeCodeProviderConfig`. Fix with `hasattr` guard:

    ```python
    def get_provider_choices() -> list[str]:
        """Build provider choices with config status indicators."""
        import shutil

        choices = []
        for name, display in _get_provider_names().items():
            provider = getattr(config.providers, name, None)
            if provider and hasattr(provider, "api_key"):
                # Standard provider -- show api_key status
                if provider.api_key:
                    choices.append(f"{display} *")
                else:
                    choices.append(display)
            elif provider and hasattr(provider, "cli_path"):
                # CLI-based provider (Claude Code) -- show CLI availability (per D-15)
                cli_available = bool(shutil.which("claude"))
                status = "installed" if cli_available else "not installed"
                choices.append(f"{display} ({status})")
            else:
                choices.append(display)
        return choices + ["<- Back"]
    ```

    IMPORTANT (Pitfall 5 from research): `shutil.which` is called inside `get_provider_choices()`, NOT inside the `lru_cache`-wrapped `_get_provider_info()`. This prevents stale cache data when the user installs claude after opening the picker.

    The existing `_configure_provider()` function at line 677 accesses `provider_config.api_base` on line 688. This will crash for `ClaudeCodeProviderConfig` which has no `api_base` field. Add a `hasattr` guard:

    ```python
    def _configure_provider(config: Config, provider_name: str) -> None:
        provider_config = getattr(config.providers, provider_name, None)
        if provider_config is None:
            console.print(f"[red]Unknown provider: {provider_name}[/red]")
            return

        display_name = _get_provider_names().get(provider_name, provider_name)
        info = _get_provider_info()
        default_api_base = info.get(provider_name, (None, None, None, None))[3]

        if default_api_base and hasattr(provider_config, "api_base") and not provider_config.api_base:
            provider_config.api_base = default_api_base

        updated_provider = _configure_pydantic_model(provider_config, display_name)
        if updated_provider is not None:
            setattr(config.providers, provider_name, updated_provider)
    ```
